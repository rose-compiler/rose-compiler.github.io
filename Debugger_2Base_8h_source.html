132 template<class Filter>
133 Sawyer::Container::Trace<rose_addr_t> trace(ThreadId tid, Filter &filter) {
134 Sawyer::Container::Trace<rose_addr_t> retval;
135 while (!isTerminated()) {
136 rose_addr_t va = executionAddress(tid);
137 FilterAction action = filter(va);
138 if (action.isClear(FilterActionFlag::REJECT))
139 retval.append(va);
140 if (action.isSet(FilterActionFlag::STOP))
141 return retval;
142 singleStep(tid);
143 }
144 return retval;
145 }
